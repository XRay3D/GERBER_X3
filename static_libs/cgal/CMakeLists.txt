set(CMAKE_AUTORCC ON)

find_package(
  Qt${QT_VERSION_MAJOR}
  COMPONENTS Widgets # OpenGL
  REQUIRED)

file(GLOB SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)

add_library(cgal STATIC ${HEADERS} ${SOURCES})
